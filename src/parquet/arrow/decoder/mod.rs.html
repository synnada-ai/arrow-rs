36// THIS ENUM IS ARAS ONLY
37///
38/// Default value for invalid UTF-8 strings.
39///
40/// This enum is used to specify the default value for invalid UTF-8 strings
41/// when decoding column values. It can be set to a specific default string,
42/// a null value, or no default value at all.
43#[derive(Debug, Default, Clone, PartialEq, Eq)]
44pub enum DefaultValueForInvalidUtf8 {
45    /// Default value for invalid UTF-8 strings.
46    Default(String),
47    /// Default value for invalid UTF-8 strings, which is a null value.
48    Null,
49    #[default]
50    /// No default value for invalid UTF-8 strings.
51    /// This means that the decoder will return an error if it encounters invalid UTF-8.
52    /// This is the default behavior.
53    None,
54}
55
56/// THIS STRUCT IS ARAS ONLY
57///
58/// Options for column value decoding behavior.
59///
60/// Contains settings that control how column values are decoded, such as
61/// whether to validate decoded values.
62///
63/// Setting `skip_validation` to true may improve performance but could
64/// result in incorrect data if the input is malformed.
65#[derive(Debug, Default, Clone)]
66pub struct ColumnValueDecoderOptions {
67    /// These two are kept separately to lower the cost of maintaining the upstream diff because UnsafeFlag is upstream feature.
68    ///
69    /// Skip validation of the values read from the column.
70    pub skip_validation: UnsafeFlag,
71    /// Default value of non-valid UTF-8 strings.
72    pub default_value: DefaultValueForInvalidUtf8,
73}
74
75impl ColumnValueDecoderOptions {
76    /// Create a new `ColumnValueDecoderOptions` with the given `skip_validation` flag.
77    pub fn new(skip_validation: UnsafeFlag, default_value: DefaultValueForInvalidUtf8) -> Self {
78        if skip_validation.get() && default_value != DefaultValueForInvalidUtf8::None {
79            // If skip_validation is set, we should not set a default value
80            // for invalid UTF-8 strings. This is because the decoder will
81            // not validate the values
82            panic!("Invalid setting");
83        }
84
85        Self {
86            skip_validation,
87            default_value,
88        }
89    }
90}
