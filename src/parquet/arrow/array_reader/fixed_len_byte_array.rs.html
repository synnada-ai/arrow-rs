52/// Returns an [`ArrayReader`] that decodes the provided fixed length byte array column
53pub fn make_fixed_len_byte_array_reader(
54    pages: Box<dyn PageIterator>,
55    column_desc: ColumnDescPtr,
56    arrow_type: Option<ArrowType>,
57) -> Result<Box<dyn ArrayReader>> {
58    // Check if Arrow type is specified, else create it from Parquet type
59    let data_type = match arrow_type {
60        Some(t) => t,
61        None => parquet_to_arrow_field(column_desc.as_ref())?
62            .data_type()
63            .clone(),
64    };
65
66    let byte_length = match column_desc.physical_type() {
67        Type::FIXED_LEN_BYTE_ARRAY => column_desc.type_length() as usize,
68        t => {
69            return Err(general_err!(
70                "invalid physical type for fixed length byte array reader - {}",
71                t
72            ))
73        }
74    };
75    match &data_type {
76        ArrowType::FixedSizeBinary(_) => {}
77        ArrowType::Decimal32(_, _) => {
78            if byte_length > 4 {
79                return Err(general_err!(
80                    "decimal 32 type too large, must be less then 4 bytes, got {}",
81                    byte_length
82                ));
83            }
84        }
85        ArrowType::Decimal64(_, _) => {
86            if byte_length > 8 {
87                return Err(general_err!(
88                    "decimal 64 type too large, must be less then 8 bytes, got {}",
89                    byte_length
90                ));
91            }
92        }
93        ArrowType::Decimal128(_, _) => {
94            if byte_length > 16 {
95                return Err(general_err!(
96                    "decimal 128 type too large, must be less than 16 bytes, got {}",
97                    byte_length
98                ));
99            }
100        }
101        ArrowType::Decimal256(_, _) => {
102            if byte_length > 32 {
103                return Err(general_err!(
104                    "decimal 256 type too large, must be less than 32 bytes, got {}",
105                    byte_length
106                ));
107            }
108        }
109        ArrowType::Interval(_) => {
110            if byte_length != 12 {
111                // https://github.com/apache/parquet-format/blob/master/LogicalTypes.md#interval
112                return Err(general_err!(
113                    "interval type must consist of 12 bytes got {}",
114                    byte_length
115                ));
116            }
117        }
118        ArrowType::Float16 => {
119            if byte_length != 2 {
120                return Err(general_err!(
121                    "float 16 type must be 2 bytes, got {}",
122                    byte_length
123                ));
124            }
125        }
126        _ => {
127            return Err(general_err!(
128                "invalid data type for fixed length byte array reader - {}",
129                data_type
130            ))
131        }
132    }
133
134    Ok(Box::new(FixedLenByteArrayReader::new(
135        pages,
136        column_desc,
137        data_type,
138        byte_length,
139    )))
140}

146    def_levels_buffer: Option<Vec<i16>>,
147    rep_levels_buffer: Option<Vec<i16>>,

149}

169impl ArrayReader for FixedLenByteArrayReader {
170    fn as_any(&self) -> &dyn Any {
171        self
172    }
173
174    fn get_data_type(&self) -> &ArrowType {
175        &self.data_type
176    }
177
178    fn read_records(&mut self, batch_size: usize) -> Result<usize> {
179        read_records(&mut self.record_reader, self.pages.as_mut(), batch_size)
180    }
181
182    fn consume_batch(&mut self) -> Result<ArrayRef> {
183        let record_data = self.record_reader.consume_record_data();
184
185        let array_data = ArrayDataBuilder::new(ArrowType::FixedSizeBinary(self.byte_length as i32))
186            .len(self.record_reader.num_values())
187            .add_buffer(Buffer::from_vec(record_data.buffer))
188            .null_bit_buffer(self.record_reader.consume_bitmap_buffer());
189
190        let binary = FixedSizeBinaryArray::from(unsafe { array_data.build_unchecked() });
191
192        // TODO: An improvement might be to do this conversion on read
193        // Note the conversions below apply to all elements regardless of null slots as the
194        // conversion lambdas are all infallible. This improves performance by avoiding a branch in
195        // the inner loop (see docs for `PrimitiveArray::from_unary`).
196        let array: ArrayRef = match &self.data_type {
197            ArrowType::Decimal32(p, s) => {
198                let f = |b: &[u8]| i32::from_be_bytes(sign_extend_be(b));
199                Arc::new(Decimal32Array::from_unary(&binary, f).with_precision_and_scale(*p, *s)?)
200                    as ArrayRef
201            }
202            ArrowType::Decimal64(p, s) => {
203                let f = |b: &[u8]| i64::from_be_bytes(sign_extend_be(b));
204                Arc::new(Decimal64Array::from_unary(&binary, f).with_precision_and_scale(*p, *s)?)
205                    as ArrayRef
206            }
207            ArrowType::Decimal128(p, s) => {
208                let f = |b: &[u8]| i128::from_be_bytes(sign_extend_be(b));
209                Arc::new(Decimal128Array::from_unary(&binary, f).with_precision_and_scale(*p, *s)?)
210                    as ArrayRef
211            }
212            ArrowType::Decimal256(p, s) => {
213                let f = |b: &[u8]| i256::from_be_bytes(sign_extend_be(b));
214                Arc::new(Decimal256Array::from_unary(&binary, f).with_precision_and_scale(*p, *s)?)
215                    as ArrayRef
216            }
217            ArrowType::Interval(unit) => {
218                // An interval is stored as 3x 32-bit unsigned integers storing months, days,
219                // and milliseconds
220                match unit {
221                    IntervalUnit::YearMonth => {
222                        let f = |b: &[u8]| i32::from_le_bytes(b[0..4].try_into().unwrap());
223                        Arc::new(IntervalYearMonthArray::from_unary(&binary, f)) as ArrayRef
224                    }
225                    IntervalUnit::DayTime => {
226                        let f = |b: &[u8]| {
227                            IntervalDayTime::new(
228                                i32::from_le_bytes(b[4..8].try_into().unwrap()),
229                                i32::from_le_bytes(b[8..12].try_into().unwrap()),
230                            )
231                        };
232                        Arc::new(IntervalDayTimeArray::from_unary(&binary, f)) as ArrayRef
233                    }
234                    IntervalUnit::MonthDayNano => {
235                        return Err(nyi_err!("MonthDayNano intervals not supported"));
236                    }
237                }
238            }
239            ArrowType::Float16 => {
240                let f = |b: &[u8]| f16::from_le_bytes(b[..2].try_into().unwrap());
241                Arc::new(Float16Array::from_unary(&binary, f)) as ArrayRef
242            }
243            _ => Arc::new(binary) as ArrayRef,
244        };
245
246        self.def_levels_buffer = self.record_reader.consume_def_levels();
247        self.rep_levels_buffer = self.record_reader.consume_rep_levels();
248        self.record_reader.reset();
249
250        Ok(array)
251    }
252
253    fn skip_records(&mut self, num_records: usize) -> Result<usize> {
254        skip_records(&mut self.record_reader, self.pages.as_mut(), num_records)
255    }
256
257    fn get_def_levels(&self) -> Option<&[i16]> {
258        self.def_levels_buffer.as_deref()
259    }
260
261    fn get_rep_levels(&self) -> Option<&[i16]> {
262        self.rep_levels_buffer.as_deref()
263    }
264}

266#[derive(Default)]
267struct FixedLenByteArrayBuffer {
268    buffer: Vec<u8>,
269    /// The length of each element in bytes
270    byte_length: Option<usize>,
271}
272
273#[inline]
274fn move_values<F>(
275    buffer: &mut Vec<u8>,
276    byte_length: usize,
277    values_range: Range<usize>,
278    valid_mask: &[u8],
279    mut op: F,
280) where
281    F: FnMut(&mut Vec<u8>, usize, usize, usize),
282{
283    for (value_pos, level_pos) in values_range.rev().zip(iter_set_bits_rev(valid_mask)) {
284        debug_assert!(level_pos >= value_pos);
285        if level_pos <= value_pos {
286            break;
287        }
288
289        let level_pos_bytes = level_pos * byte_length;
290        let value_pos_bytes = value_pos * byte_length;
291
292        op(buffer, level_pos_bytes, value_pos_bytes, byte_length)
293    }
294}

296impl ValuesBuffer for FixedLenByteArrayBuffer {
297    fn pad_nulls(
298        &mut self,
299        read_offset: usize,
300        values_read: usize,
301        levels_read: usize,
302        valid_mask: &[u8],
303    ) {
304        let byte_length = self.byte_length.unwrap_or_default();
305
306        assert_eq!(self.buffer.len(), (read_offset + values_read) * byte_length);
307        self.buffer
308            .resize((read_offset + levels_read) * byte_length, 0);
309
310        let values_range = read_offset..read_offset + values_read;
311        // Move the bytes from value_pos to level_pos. For values of `byte_length` <= 4,
312        // the simple loop is preferred as the compiler can eliminate the loop via unrolling.
313        // For `byte_length > 4`, we instead copy from non-overlapping slices. This allows
314        // the loop to be vectorized, yielding much better performance.
315        const VEC_CUTOFF: usize = 4;
316        if byte_length > VEC_CUTOFF {
317            let op = |buffer: &mut Vec<u8>, level_pos_bytes, value_pos_bytes, byte_length| {
318                let split = buffer.split_at_mut(level_pos_bytes);
319                let dst = &mut split.1[..byte_length];
320                let src = &split.0[value_pos_bytes..value_pos_bytes + byte_length];
321                dst.copy_from_slice(src);
322            };
323            move_values(&mut self.buffer, byte_length, values_range, valid_mask, op);
324        } else {
325            let op = |buffer: &mut Vec<u8>, level_pos_bytes, value_pos_bytes, byte_length| {
326                for i in 0..byte_length {
327                    buffer[level_pos_bytes + i] = buffer[value_pos_bytes + i]
328                }
329            };
330            move_values(&mut self.buffer, byte_length, values_range, valid_mask, op);
331        }
332    }
333}

335struct ValueDecoder {
336    byte_length: usize,
337    dict_page: Option<Bytes>,
338    decoder: Option<Decoder>,
339}
340
341impl ColumnValueDecoder for ValueDecoder {
342    type Buffer = FixedLenByteArrayBuffer;
343
344    fn new(col: &ColumnDescPtr) -> Self {
345        Self {
346            byte_length: col.type_length() as usize,
347            dict_page: None,
348            decoder: None,
349        }
350    }
351
352    /// THIS METHOD IS ARAS ONLY
353    fn new_with_options(
354        _options: ColumnValueDecoderOptions,
355        col: &ColumnDescPtr,
356        _data_type: ArrowType,
357    ) -> Self {
358        Self::new(col)
359    }
360
361    fn set_dict(
362        &mut self,
363        buf: Bytes,
364        num_values: u32,
365        encoding: Encoding,
366        _is_sorted: bool,
367    ) -> Result<()> {
368        if !matches!(
369            encoding,
370            Encoding::PLAIN | Encoding::RLE_DICTIONARY | Encoding::PLAIN_DICTIONARY
371        ) {
372            return Err(nyi_err!(
373                "Invalid/Unsupported encoding type for dictionary: {}",
374                encoding
375            ));
376        }
377        let expected_len = num_values as usize * self.byte_length;
378        if expected_len > buf.len() {
379            return Err(general_err!(
380                "too few bytes in dictionary page, expected {} got {}",
381                expected_len,
382                buf.len()
383            ));
384        }
385
386        self.dict_page = Some(buf);
387        Ok(())
388    }
389
390    fn set_data(
391        &mut self,
392        encoding: Encoding,
393        data: Bytes,
394        num_levels: usize,
395        num_values: Option<usize>,
396    ) -> Result<()> {
397        self.decoder = Some(match encoding {
398            Encoding::PLAIN => Decoder::Plain {
399                buf: data,
400                offset: 0,
401            },
402            Encoding::RLE_DICTIONARY | Encoding::PLAIN_DICTIONARY => Decoder::Dict {
403                decoder: DictIndexDecoder::new(data, num_levels, num_values),
404            },
405            Encoding::DELTA_BYTE_ARRAY => Decoder::Delta {
406                decoder: DeltaByteArrayDecoder::new(data)?,
407            },
408            Encoding::BYTE_STREAM_SPLIT => Decoder::ByteStreamSplit {
409                buf: data,
410                offset: 0,
411            },
412            _ => {
413                return Err(general_err!(
414                    "unsupported encoding for fixed length byte array: {}",
415                    encoding
416                ))
417            }
418        });
419        Ok(())
420    }
421
422    fn read(&mut self, out: &mut Self::Buffer, num_values: usize) -> Result<usize> {
423        match out.byte_length {
424            Some(x) => assert_eq!(x, self.byte_length),
425            None => out.byte_length = Some(self.byte_length),
426        }
427
428        match self.decoder.as_mut().unwrap() {
429            Decoder::Plain { offset, buf } => {
430                let to_read =
431                    (num_values * self.byte_length).min(buf.len() - *offset) / self.byte_length;
432                let end_offset = *offset + to_read * self.byte_length;
433                out.buffer
434                    .extend_from_slice(&buf.as_ref()[*offset..end_offset]);
435                *offset = end_offset;
436                Ok(to_read)
437            }
438            Decoder::Dict { decoder } => {
439                let dict = self.dict_page.as_ref().unwrap();
440                // All data must be NULL
441                if dict.is_empty() {
442                    return Ok(0);
443                }
444
445                decoder.read(num_values, |keys| {
446                    out.buffer.reserve(keys.len() * self.byte_length);
447                    for key in keys {
448                        let offset = *key as usize * self.byte_length;
449                        let val = &dict.as_ref()[offset..offset + self.byte_length];
450                        out.buffer.extend_from_slice(val);
451                    }
452                    Ok(())
453                })
454            }
455            Decoder::Delta { decoder } => {
456                let to_read = num_values.min(decoder.remaining());
457                out.buffer.reserve(to_read * self.byte_length);
458
459                decoder.read(to_read, |slice| {
460                    if slice.len() != self.byte_length {
461                        return Err(general_err!(
462                            "encountered array with incorrect length, got {} expected {}",
463                            slice.len(),
464                            self.byte_length
465                        ));
466                    }
467                    out.buffer.extend_from_slice(slice);
468                    Ok(())
469                })
470            }
471            Decoder::ByteStreamSplit { buf, offset } => {
472                // we have n=`byte_length` streams of length `buf.len/byte_length`
473                // to read value i, we need the i'th byte from each of the streams
474                // so `offset` should be the value offset, not the byte offset
475                let total_values = buf.len() / self.byte_length;
476                let to_read = num_values.min(total_values - *offset);
477
478                // now read the n streams and reassemble values into the output buffer
479                read_byte_stream_split(&mut out.buffer, buf, *offset, to_read, self.byte_length);
480
481                *offset += to_read;
482                Ok(to_read)
483            }
484        }
485    }
486
487    /// THIS METHOD IS ARAS ONLY
488    fn read_with_null_mask(
489        &mut self,
490        out: &mut Self::Buffer,
491        num_values: usize,
492    ) -> Result<Vec<bool>> {
493        let len = self.read(out, num_values)?;
494        Ok(vec![true; len])
495    }
496
497    fn skip_values(&mut self, num_values: usize) -> Result<usize> {
498        match self.decoder.as_mut().unwrap() {
499            Decoder::Plain { offset, buf } => {
500                let to_read = num_values.min((buf.len() - *offset) / self.byte_length);
501                *offset += to_read * self.byte_length;
502                Ok(to_read)
503            }
504            Decoder::Dict { decoder } => decoder.skip(num_values),
505            Decoder::Delta { decoder } => decoder.skip(num_values),
506            Decoder::ByteStreamSplit { offset, buf } => {
507                let total_values = buf.len() / self.byte_length;
508                let to_read = num_values.min(total_values - *offset);
509                *offset += to_read;
510                Ok(to_read)
511            }
512        }
513    }
514}

516// `src` is an array laid out like a NxM matrix where N == `data_width` and
517// M == total_values_in_src. Each "row" of the matrix is a stream of bytes, with stream `i`
518// containing the `ith` byte for each value. Each "column" is a single value.
519// This will reassemble `num_values` values by reading columns of the matrix starting at
520// `offset`. Values will be appended to `dst`.
521fn read_byte_stream_split(
522    dst: &mut Vec<u8>,
523    src: &mut Bytes,
524    offset: usize,
525    num_values: usize,
526    data_width: usize,
527) {
528    let stride = src.len() / data_width;
529    let idx = dst.len();
530    dst.resize(idx + num_values * data_width, 0u8);
531    let dst_slc = &mut dst[idx..idx + num_values * data_width];
532    for j in 0..data_width {
533        let src_slc = &src[offset + j * stride..offset + j * stride + num_values];
534        for i in 0..num_values {
535            dst_slc[i * data_width + j] = src_slc[i];
536        }
537    }
538}

540enum Decoder {
541    Plain { buf: Bytes, offset: usize },
542    Dict { decoder: DictIndexDecoder },
543    Delta { decoder: DeltaByteArrayDecoder },
544    ByteStreamSplit { buf: Bytes, offset: usize },
545}

547#[cfg(test)]
548mod tests {
549    use super::*;
550    use crate::arrow::arrow_reader::ParquetRecordBatchReader;
551    use crate::arrow::ArrowWriter;
552    use arrow::datatypes::Field;
553    use arrow::error::Result as ArrowResult;
554    use arrow_array::{Array, ListArray};
555    use arrow_array::{Decimal256Array, RecordBatch};
556    use bytes::Bytes;
557    use std::sync::Arc;
558
559    #[test]
560    fn test_decimal_list() {
561        let decimals = Decimal256Array::from_iter_values(
562            [1, 2, 3, 4, 5, 6, 7, 8].into_iter().map(i256::from_i128),
563        );
564
565        // [[], [1], [2, 3], null, [4], null, [6, 7, 8]]
566        let data = ArrayDataBuilder::new(ArrowType::List(Arc::new(Field::new_list_field(
567            decimals.data_type().clone(),
568            false,
569        ))))
570        .len(7)
571        .add_buffer(Buffer::from_iter([0_i32, 0, 1, 3, 3, 4, 5, 8]))
572        .null_bit_buffer(Some(Buffer::from(&[0b01010111])))
573        .child_data(vec![decimals.into_data()])
574        .build()
575        .unwrap();
576
577        let written =
578            RecordBatch::try_from_iter([("list", Arc::new(ListArray::from(data)) as ArrayRef)])
579                .unwrap();
580
581        let mut buffer = Vec::with_capacity(1024);
582        let mut writer = ArrowWriter::try_new(&mut buffer, written.schema(), None).unwrap();
583        writer.write(&written).unwrap();
584        writer.close().unwrap();
585
586        let read = ParquetRecordBatchReader::try_new(Bytes::from(buffer), 3)
587            .unwrap()
588            .collect::<ArrowResult<Vec<_>>>()
589            .unwrap();
590
591        assert_eq!(&written.slice(0, 3), &read[0]);
592        assert_eq!(&written.slice(3, 3), &read[1]);
593        assert_eq!(&written.slice(6, 1), &read[2]);
594    }
